import numpy as np
import random

def fitness(population, n, t):
    fitpoint = []
    for chromosome in population:
        overpenalty = 0
        consist_penalty = 0

        for time in range(t):
            courses_in_slot = sum(chromosome[time*n : (time+1)*n])
            if courses_in_slot > 1:
                overpenalty += courses_in_slot - 1

        for course in range(n):
            times_scheduled = sum(chromosome[course::n])
            if times_scheduled != 1:


                consist_penalty += abs(times_scheduled - 1)

        point = -(overpenalty + consist_penalty)
        fitpoint.append(point)

    return np.array(fitpoint)

def selection(population, fitpoint):
    probabilities = fitpoint / fitpoint.sum()
    return population[np.random.choice(len(population), p=probabilities)]

def crossover(p1, p2):
    point = random.randint(1, len(p1) - 1)
    child1 = np.concatenate((p1[:point], p2[point:]))


    child2 = np.concatenate((p2[:point], p1[point:]))
    return child1, child2

def mutate(chromosome, mutation_rate):
    for i in range(len(chromosome)):
        if random.random() < mutation_rate:
            chromosome[i] = 1 - chromosome[i]
    return chromosome

def genetic_algorithm(n, t, population_size=100, generations=1000, mutation_rate=0.01):
    population = np.random.randint(2, size=(population_size, n * t))

    for generation in range(generations):
        fitpoint = fitness(population, n, t)
        new_population = []

        for _ in range(population_size // 2):
            p1 = selection(population, fitpoint)
            p2 = selection(population, fitpoint)

            child1, child2 = crossover(p1, p2)

            child1 = mutate(child1, mutation_rate)
            child2 = mutate(child2, mutation_rate)



            new_population.extend([child1, child2])

        population = np.array(new_population)

    final_fitpoint = fitness(population, n, t)
    best_index = np.argmax(final_fitpoint)
    best_chromosome = population[best_index]
    best_fitness = final_fitpoint[best_index]

    print("".join(map(str, best_chromosome)))
    print(best_fitness)
