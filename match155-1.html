import random





def fitness_function(new_choromosome, N, T):

    open_loop_penalty = 0
    closed_system_cost = 0



    for j in range(T):  # penalty overlap
        courses = sum(int(new_choromosome[j * N + i]) for i in range(N))


        if courses > 1:
            open_loop_penalty += courses - 1


    for k in range(N):  # penalty for consistency
        time_of_mutaion = sum(int(new_choromosome[i * N + k]) for i in range(T))
        closed_system_cost += abs(time_of_mutaion - 1)


    return -(open_loop_penalty + closed_system_cost)


def initialize_population(population_xount, given_chorm):  # initial population for chromosomes

    return [''.join(random.choice('01') for l in range(given_chorm)) for m in range(population_xount)]


def select_parents(ranodome_gene):  # population parents

    return random.sample(ranodome_gene, 2)


def genetic_algorithm(N, T, size_p=10, gen_high=1000, val_of_mutation=0.01, main_chro=None):  # Genetic Algorithm



    given_chorm = N * T

    ranodome_gene = initialize_population(size_p, given_chorm)


    for x in range(gen_high):

        storing_population = []


        for c in range(size_p // 2):

            par_one, par_two = select_parents(ranodome_gene)
            ofss_one, ofss_two = two_point_crossover(par_one, par_two)
            ofss_one = mutate(ofss_one, val_of_mutation)
            ofss_two = mutate(ofss_two, val_of_mutation)

            storing_population.extend([ofss_one, ofss_two])

        ranodome_gene = sorted(storing_population, key=lambda x: fitness_function(x, N, T), reverse=True)[:size_p]


        if main_chro and main_chro in ranodome_gene:

            return main_chro, fitness_function(main_chro, N, T)

        #print(f"Generation: {x}, Best fitness: {fitness_function(max(ranodome_gene, key=lambda x: fitness_function(x, N, T)), N, T)}")


    ans = max(ranodome_gene, key=lambda x: fitness_function(x, N, T))
    return ans, fitness_function(ans, N, T)


def mutate(new_choromosome, val_of_mutation=0.01):  # Mutation
    chr_list = list(new_choromosome)
    for i in range(len(chr_list)):
        if random.random() < val_of_mutation:
            chr_list[i] = '1' if chr_list[i] == '0' else '0'
    return ''.join(chr_list)


def single_point_crossover(par_one, par_two):  # crossover single
    cross_over = random.randint(1, len(par_one) - 1)
    ofss_one = par_one[:cross_over] + par_two[cross_over:]
    ofss_two = par_two[:cross_over] + par_one[cross_over:]
    return ofss_one, ofss_two


def two_point_crossover(par_one, par_two):  # crossover 2 position
    mutd_one = random.randint(1, len(par_one) - 2)
    mutd_two = random.randint(mutd_one + 1, len(par_one) - 1)
    ofss_one = par_one[:mutd_one] + par_two[mutd_one:mutd_two] + par_one[mutd_two:]
    ofss_two = par_two[:mutd_one] + par_one[mutd_one:mutd_two] + par_two[mutd_two:]
    return ofss_one, ofss_two

import random

def two_point_crossover(prn_one, pnr_two):

  crsvr_one = random.randint(0, len(prn_one) - 2)
  crsve_two = random.randint(crsvr_one + 1, len(prn_one) - 1)


  ofss_one = prn_one[:crsvr_one] + pnr_two[crsvr_one:crsve_two] + prn_one[crsve_two:]
  ofss_two = pnr_two[:crsvr_one] + prn_one[crsvr_one:crsve_two] + pnr_two[crsve_two:]

  return ofss_one, ofss_two
