#Part-1

import random
#Parameters


POP_SIZE =10
MUT_RATE =0.01
MAX_GEN =1000


def create_chromosomes():                                        # Chromosome Representation
    chromosomes = [0]*(num_courses * num_timeslots)
    course_indices =list(range(num_courses))
    random.shuffle(course_indices)

    for idx,course_idx in enumerate(course_indices):
        timeslot =idx%num_timeslots


        chromosomes[timeslot * num_courses+course_idx] =  1
    return chromosomes

def fitness_function(chromosomes):
    overlap_penalty= 0
    consistency_penalty= 0
    # Check for overlaps in timeslots
    for t in range(num_timeslots):
        scheduled_courses =sum(chromosomes[t*num_courses:num_courses*(t+1)])
        if scheduled_courses>1:
            overlap_penalty +=(scheduled_courses-1)

    # Check for course scheduling consistency
    for a in range(num_courses):
        scheduled_times = sum([chromosomes[t*num_courses +a] for t in range(num_timeslots)])
        if scheduled_times!=1:
            consistency_penalty += abs(scheduled_times-1)
    return  -(consistency_penalty+overlap_penalty)

#  Parent Selection


def choose_parents(population):
    fitness_values = [fitness_function(chromosomes) for chromosomes in population]
    total_fitness = sum(fitness_values)
    if total_fitness == 0:
        return random.choices(population, k=2)
    selection_probs = [fitness / total_fitness for fitness in fitness_values]
    return random.choices(population, weights=selection_probs, k=2)

# Crossover
def crossover(parent1,parent2):
    crossover_point =random.randint(1,num_courses*num_timeslots-2)
    offspring1 =parent1[:crossover_point]+parent2[crossover_point:]


    offspring2 =parent2[:crossover_point]+parent1[crossover_point:]

    return offspring1,offspring2

# Mutation
def mutation(chromosomes):
    for i in range(len(chromosomes)):
        if random.random()<MUT_RATE:
            chromosomes[i]=1 if chromosomes[i] == 0 else 0
    return chromosomes



population = [create_chromosomes() for i in range(POP_SIZE)]

# Genetic
for generation in range(MAX_GEN):
    new_population =[]
    for i in range(POP_SIZE //2):
        parent1, parent2 = choose_parents(population)
        child1, child2 = crossover(parent1, parent2)
        new_population.append(mutation(child1))
        new_population.append(mutation(child2))
    population = sorted(new_population,key=fitness_function, reverse=True)[:POP_SIZE]

best_chromosome = max(population, key=fitness_function)
best_fitness_value =fitness_function(best_chromosome)
print("Best Schedule (Binary String):",''.join(map(str,best_chromosome)))
print("Fitness Value:",best_fitness_value)
