#Part 1

import random

def fitness_calculation(chromosome, N, T):
    overlap = 0
    consistency = 0

    schedule = [chromosome[i * N:(i + 1) * N] for i in range(T)]



    for t in schedule:
        s_courses = sum(t)
        if s_courses > 1:
            overlap += (s_courses - 1)

    for i in range(N):
        number_of_courses = sum(t[i] for t in schedule)
        consistency += abs(number_of_courses - 1)

    total = overlap + consistency
    Fitness = -total
    return Fitness

N, T = map(int, input().split())
courses = []
for i in range(N):


  courses.append(input().strip())

chromosome = [random.randint(0, 1) for i in range(N * T)]

while all(value == 0 for value in chromosome):
    chromosome = [random.randint(0, 1) for i in range(N * T)]

fitness = fitness_calculation(chromosome, N, T)

output = (
    f"Chromosome: {''.join(map(str, chromosome))}\n"
    f"Fitness Value: {fitness}\n"
)
with open("output1.txt", "w") as file:
    file.write(output)

import random

def two_point_crossover(Parent_1, Parent_2):
    assert len(Parent_1) == len(Parent_2)

    length = len(Parent_1)

    Point1 = random.randint(0, length - 2)
    Point2 = random.randint(Point1 + 1, length - 1)

    os1 = Parent_1[:Point1] + Parent_2[Point1:Point2] + Parent_1[Point2:]


    os2 = Parent_2[:Point1] + Parent_1[Point1:Point2] + Parent_2[Point2:]

    return os1, os2, Point1, Point2
