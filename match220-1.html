import random

with open("input.txt", "r") as file:
    line = file.readline().split()
    num_course = int(line[0])
    num_slots = int(line[1])

print("--------------------Part1--------------------")
chromosome_length = num_course * num_slots
population = []

for i in range(10):
    chromosome = []
    for j in range(chromosome_length):
        gene = random.choice([0, 1])
        chromosome.append(gene)
    population.append(chromosome)

def fitness_overlap_function(chromosome):
    total_overlap = 0
    for i in range(0, len(chromosome), num_course):
        slot_courses = chromosome[i:i + num_course]
        sum_slot_courses = sum(slot_courses)
        if sum_slot_courses > 1:
            total_overlap += sum_slot_courses - 1
    return total_overlap


def fitness_consistency_function(chromosome):
    total_inconsistency = 0
    for j in range(num_course):
        course_occured = sum(chromosome[j::num_course])
        total_inconsistency += abs(course_occured - 1)
    return total_inconsistency

def fitness_function(chromosome):


    return -(fitness_overlap_function(chromosome) + fitness_consistency_function(chromosome))

def parent_selection(population):
    parent1,parent2 = random.sample(population, 2)
    return parent1, parent2

def crossover(parent1, parent2):
    crossover_point = random.randint(1, chromosome_length - 1)
    child1 = parent1[:crossover_point] + parent2[crossover_point:]
    child2 = parent2[:crossover_point] + parent1[crossover_point:]
    return child1, child2

def mutate(chromosome):
    mutation_point = random.randint(0, chromosome_length - 1)
    chromosome[mutation_point] = 1 - chromosome[mutation_point]
    return chromosome

def reformulation(population):
    reformated_list = []
    for i in range(len(population)):
        fitness = fitness_function(population[i])
        reformated_list.append((fitness, population[i]))
    reformated_list.sort(reverse=True)
    population=[]
    for i in range(10):
        population.append(reformated_list[i][1])
    return population

def genetic_algorithm(population):
    for iteration in range(100):


        parent1, parent2 = parent_selection(population)

        if fitness_function(parent1) == 0:
            return parent1
        elif fitness_function(parent2) == 0:
            return parent2

        child1, child2 = crossover(parent1, parent2)
        child1 = mutate(child1)
        child2 = mutate(child2)

        population.append(child1)
        population.append(child2)
        population = reformulation(population)

    return population[0]

best_solution = genetic_algorithm(population)

for i in range(len(best_solution)):
    print(best_solution[i],end='')
print("\nFitness:", fitness_function(best_solution))
