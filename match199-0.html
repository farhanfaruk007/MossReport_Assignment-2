def calculate_fitness(chromosome, N, T):


    timeslots = [chromosome[i * N:(i + 1) * N] for i in range(T)]

    overlap_penalty = 0
    for t in timeslots:
        course_scheduled = sum(t)
        if course_scheduled > 1:
          overlap_penalty += (course_scheduled - 1)

    course_count = [0] * N
    for t in timeslots:
        for i in range(N):
            course_count[i] += t[i]

    penalty_c = sum(abs(count - 1) for count in course_count)

    total_penalty = overlap_penalty + penalty_c

    fitness = -total_penalty
    return fitness

import random

def initial_pop(pop_size, N, T):
    population = []
    length_of_chromosome = N * T

    for i in range(pop_size):


        chromosome = [random.randint(0, 1) for i in range(length_of_chromosome)]
        if sum(chromosome) == 0:
            chromosome[random.randint(0, length_of_chromosome - 1)] = 1
        population.append(chromosome)

    return population

def two_point_crossover(parent_1, parent_2):
    length = len(parent_1)

    point_1 = random.randint(1, length - 2)
    point_2 = random.randint(point_1 + 1, length - 1)

    offspring_1 = parent_1[:point_1] + parent_2[point_1:point_2] + parent_1[point_2:]
    offspring_2 = parent_2[:point_1] + parent_1[point_1:point_2] + parent_2[point_2:]

    return offspring_1, offspring_2, (point_1, point_2)
