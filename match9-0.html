# Fitness Function :
def fitness_fun_overlap(chr) :
  sum = 0
  slot = []
  # overlap & no course selected
  for i in range(0,len(chr),T) :
    for j in range(N) :
      sum += chr[i+j]
      # print("SUM",sum)
    slot.append(sum-1)
    sum = 0
  a = 0
  for i in range(T) :
    if slot[i] != -1 :
      a += slot[i]
  return a


def fitness_fun_Consistency(chr) :
  sum = 0
  slot = []
  # Consistency & one course never selected
  for j in range(N) :
    for i in range(0,len(chr),T) :
      sum += chr[i+j]
    slot.append(abs(sum-1))
    sum = 0
  a = 0
  for i in range(T) :
    a += slot[i]
  return a


def fitness_fun(chr):
  return (-1)*(fitness_fun_Consistency(chr)+fitness_fun_overlap(chr))
def selection(population):
  parent1,parent2 = random.sample(population,2)
  return parent1,parent2

def crossover(p1,p2):

  for j in range(point,N*T) :

  return child1,child2

def mutation(a,b) :
  # for 1 :
  point1 = random.randint(0,chr_l-1)
  a[point1] = random.choice(li_choice)
  # for 2 :
  point2 = random.randint(0,chr_l-1)
  b[point2] = random.choice(li_choice)
  return a,b


def reformat(population) :
  re_li = []
  for i in range(len(population)) :
    x = population[i]
    re_li.append((fitness_fun(x),x))
  re_li.sort(reverse=True)
  population = []
  for i in range(len(population)) :


    population.append(re_li[i])




def Genetic_Algo(population) :
  p1,p2 = selection(population)
  if fitness_fun(p1) == 0 :
    return p1
  elif  fitness_fun(p2) == 0 :
    return p2
  ch1,ch2 = crossover(p1,p2)
  m_ch1,m_ch2 = mutation(ch1,ch2)
  population.append(m_ch1)
  population.append(m_ch2)

  reformat(population)

  return None
