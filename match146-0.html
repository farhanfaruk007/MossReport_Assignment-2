import random

def generate_initial_population(population_size, num_courses, num_timeslots):
    population = []
    for _ in range(population_size):
        individual = []


        for _ in range(num_courses):
            timeslot = [0] * num_timeslots
            timeslot[random.randint(0, num_timeslots - 1)] = 1
            individual.extend(timeslot)
        population.append(individual)
    return population

def calculate_fitness(individual, num_courses, num_timeslots):
    overlap_penalty = 0
    consistency_penalty = 0

    timetable = [individual[i * num_timeslots:(i + 1) * num_timeslots] for i in range(num_courses)]

    for t in range(num_timeslots):
        scheduled_courses = sum(timetable[i][t] for i in range(num_courses))
        if scheduled_courses > 1:
            overlap_penalty += (scheduled_courses - 1)

    for i in range(num_courses):
        scheduled_count = sum(timetable[i])
        consistency_penalty += abs(scheduled_count - 1)

    return -(overlap_penalty + consistency_penalty)

def select_parents(population, fitness_scores):
    total_fitness = sum(fitness_scores)
    probabilities = [score / total_fitness for score in fitness_scores]
    selected_parents = random.choices(population, weights=probabilities, k=2)
    return selected_parents

def perform_crossover(parent_a, parent_b):
    crossover_point = random.randint(1, len(parent_a) - 1)
    child_a = parent_a[:crossover_point] + parent_b[crossover_point:]
    child_b = parent_b[:crossover_point] + parent_a[crossover_point:]
    return child_a, child_b

def apply_mutation(individual, mutation_rate):
    for i in range(len(individual)):
        if random.random() < mutation_rate:
            individual[i] = 1 - individual[i]


    return individual

def run_genetic_algorithm(num_courses, num_timeslots, course_names, population_size=50, max_generations=100, mutation_rate=0.02):
    population = generate_initial_population(population_size, num_courses, num_timeslots)

    for generation in range(max_generations):
        fitness_values = [calculate_fitness(individual, num_courses, num_timeslots) for individual in population]

        if 0 in fitness_values:
            break

        next_generation = []
        while len(next_generation) < population_size:
            parent_a, parent_b = select_parents(population, fitness_values)
            offspring_a, offspring_b = perform_crossover(parent_a, parent_b)
            next_generation.append(apply_mutation(offspring_a, mutation_rate))
            next_generation.append(apply_mutation(offspring_b, mutation_rate))

        population = next_generation[:population_size]

    best_individual_index = fitness_values.index(max(fitness_values))
    print(population[best_individual_index])
    print("Fitness score:", fitness_values[best_individual_index])

# Example usage
courses = ["CSE110", "MAT110", "PHY112"]
run_genetic_algorithm(3, 3, courses)
