def fitness_calculation(chromosome,t,n):
  consistency = [0] * n
	penalty = 0

  for i in range t:
    overlapping = 0
  for j in range n:
    index=n*i+j
    if chromosome[index]==1:
      overlapping+=1
      consistency[j]+=1
    if overlapping>1:
      penalty=penalty + overlapping-1
  for k in consistency:
    penalty=penalty + abs(k-1)

  return -penalty
