# Fitness Function :
def fitness_fun_overlap(chromosome) :
  sum = 0
  slot = []
  # overlap & no course selected
  for i in range(0,len(chromosome),T) :
    for j in range(N) :
      sum += chromosome[i+j]
      # print("SUM",sum)
    slot.append(sum-1)
    sum = 0
  a = 0
  for i in range(T) :
    if slot[i] != -1 :
      a += slot[i]
  return a


def fitness_fun_Consistency(chromosome) :
  sum = 0
  slot = []
  # Consistency & one course never selected
  for j in range(N) :
    for i in range(0,len(chromosome),T) :
      sum += chromosome[i+j]
    slot.append(abs(sum-1))
    sum = 0
  a = 0
  for i in range(T) :
    a += slot[i]
  return a


def fitness_fun(chromosome):
  return (-1)*(fitness_fun_Consistency(chromosome)+fitness_fun_overlap(chromosome))
def selection(population):
  parent1,parent2 = random.sample(population,2)
  return parent1,parent2

def crossover(p1,p2):
  child1 = []
  child2 = []
  point = random.randint(0,chromosome_lenght-1)
  # print(point)

  for i in range(0,point) :

  for j in range(point,N*T) :
    child1.append(p2[j])
  #child2
  for i in range(0,point) :
    child2.append(p2[i])
  for j in range(point,N*T) :
    child2.append(p1[j])
  # print(child1)
  # print(child2)
  return child1,child2

def mutation(a,b) :
  # for 1 :

  return a,b


def reformat(population) :

def Genetic_Algo(population) :
  p1,p2 = selection(population)
  if fitness_fun(p1) == 0 :
    return p1
  elif  fitness_fun(p2) == 0 :
    return p2
  ch1,ch2 = crossover(p1,p2)
  m_ch1,m_ch2 = mutation(ch1,ch2)
  population.append(m_ch1)
  population.append(m_ch2)

  reformat(population)

  return None
