def fitness(A):
  overlap = 0
  consistent = 0

  for i in range(0, len(A), N):
    sum = 0
    for course in A[i: i+N]:
      sum += course



    if sum> 1:
      overlap += (sum - 1)

  # print(overlap)

  for i in range(len(A)):
    idx = i % N
    courses[idx][1] += A[i]

  for a in courses:


    consistent += abs(a[1] - 1)

  return -1 * (overlap + consistent)

def mutation(A, B):
  idx1 = random.randint(0, len(A)-1)
  idx2 = random.randint(0, len(B)-1)

  if A[idx1] == 0:
    A[idx1] = 1
  else:
    A[idx1] = 0

  if B[idx2] == 0:
    B[idx2] = 1
  else:
    B[idx2] = 0

  return [A,B]
