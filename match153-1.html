import random

def chromoMaker(t, c):
    s_length = t * c
    chromo = [random.randint(0, 1) for _ in range(s_length)]
    return ''.join(map(str, chromo))

def populationMaker(population_size, t, c):
    return [chromoMaker(t, c) for i in range(population_size)]

def penaltyChecker(chromosome, t, c):
    penalty_time = 0
    segment = []
    count = 0

    for i in range(len(chromosome)):
        segment.append(chromosome[i])
        count += 1

        if count == t:
            classes_in_time_slot = sum(int(bit) for bit in segment)
            if classes_in_time_slot > 1:
                penalty_time += 1

            count = 0
            segment = []

    penalty_course = 0
    for i in range(t):
        course_segment = []

        for j in range(i, len(chromosome), t):
            course_segment.append(chromosome[j])

        classes_in_course = sum(int(bit) for bit in course_segment)
        penalty_course += abs(classes_in_course - 1)

    total_penalty = -(penalty_time + penalty_course)


    return total_penalty

    selected1 = population[idx1]

    return selected1, selected2

def single_point_crossover(p1, p2):
    c_point = random.randint(1, len(p1) - 1)
    offspring1 = p1[:c_point] + p2[c_point:]
    offspring2 = p2[:c_point] + p1[c_point:]
    return offspring1, offspring2

mutation_rate = random.randint(1, 99)
mutation_threshold = 60

def mutate(chromosome):
    mutated = ""
    for bit in chromosome:
        if mutation_rate > mutation_threshold:
            mutated += "1" if bit == "0" else "0"
        else:
            mutated += bit
    return mutated

def genetic_algo(population, t, c):
    max_fitness = 0
    max_iterations = 50
    iteration = 0

    current_population = population

    while iteration < max_iterations:

        fitness_scores = [penaltyChecker(chromosome, t, c) for chromosome in current_population]
        fittest_index = fitness_scores.index(max(fitness_scores))
        fittest_chromosome = current_population[fittest_index]

        if fitness_scores[fittest_index] >= max_fitness:
            break

        parent1, parent2 = randomSelection(current_population)

        offspring1, offspring2 = single_point_crossover(parent1, parent2)
        offspring1 = mutate(offspring1)
        offspring2 = mutate(offspring2)

        least_fit_indices = sorted(range(len(fitness_scores)), key=lambda x: fitness_scores[x])[:2]
        current_population[least_fit_indices[0]] = offspring1
        current_population[least_fit_indices[1]] = offspring2

        iteration += 1

    fitness_scores = [penaltyChecker(chromosome, t, c) for chromosome in current_population]
    fittest_index = fitness_scores.index(max(fitness_scores))
    return current_population[fittest_index], fitness_scores[fittest_index]

        raise ValueError("Both parent chromosomes must have the same length.")

    length = len(parent1)

    offspring1 = parent1[:point1] + parent2[point1:point2] + parent1[point2:]


    offspring2 = parent2[:point1] + parent1[point1:point2] + parent2[point2:]

    return offspring1, offspring2
