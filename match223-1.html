import random

def cei(nn, oo):
    return ''.join(random.choice('01') for _ in range(nn * oo))

def calu_ssj(cho, nn, oo):
    oap_ltyp = 0


    hh = [0] * nn

    for mmo in range(oo):
        pe = cho[mmo * nn:(mmo + 1) * nn]
        sdses = pe.count('1')
        oap_ltyp += max(0, sdses - 1)

        for kl, yu in enumerate(pe):
            hh[kl] += int(yu)

    nsis_lty = sum(abs(ct - 1) for ct in hh)

    return -(oap_ltyp + nsis_lty)

def gestn(tnize, nn, oo):
    tn = []
    while len(tn) < tnize:
        cho = cei(nn, oo)
        if calu_ssj(cho, nn, oo) != 0:
            tn.append(cho)
    return tn

def ectlents(tn, nn, oo):
    tedtn = sorted(tn, key=lambda chromo: calu_ssj(chromo, nn, oo), reverse=True)
    return tedtn[0], tedtn[1]

def ssovec(paty1, paty2):
    ssovec_oit = random.randint(0, len(paty1) - 1)
    cldo1 = paty1[:ssovec_oit] + paty2[ssovec_oit:]
    cldo2 = paty2[:ssovec_oit] + paty1[ssovec_oit:]
    return cldo1, cldo2

def yutu(cho, yututr=0.01):
    return ''.join(
        '1' if bit == '0' and random.random() < yututr else
        '0' if bit == '1' and random.random() < yututr else bit
        for bit in cho
    )

def gs_alm(nn, oo, tnize=10, max_gs=1000, yututr=0.01):
    tn = gestn(tnize, nn, oo)

    for ges in range(max_gs):
        ntn = []

        while len(ntn) < tnize:
            paty1, paty2 = ectlents(tn, nn, oo)
            cld1, cld2 = ssovec(paty1, paty2)
            ntn.append(yutu(cld1, yututr))
            ntn.append(yutu(cld2, yututr))

        tn = sorted(ntn, key=lambda chromo: calu_ssj(chromo, nn, oo), reverse=True)
        estb_choe = tn[0]
        estb_nesf = calu_ssj(estb_choe, nn, oo)

        if estb_nesf == 0:
            return estb_choe, estb_nesf, ges + 1

    return tn[0], calu_ssj(tn[0], nn, oo), max_gs

def rif(il_o):
    with open(il_o, 'r') as file:
        lines = file.readlines()



    nn, oo = map(int, lines[0].strip().split())
    sesc = [line.strip() for line in lines[1:]]
    return nn, oo, sesc

def intrlt(estb_choe, estb_nesf, gs):
    print( estb_choe)
    print("fitnessofit:", estb_nesf)
    print("GeneratiosUsed:", gs)

def main(npo):
    nn, oo , _ = rif(npo)
    estb_choe, estb_nesf, gs = gs_alm(nn, oo)
    intrlt(estb_choe, estb_nesf, gs)

npo = 'inamina.txt'
main(npo)

import random
def choe(tyuy, ioi):
    return ''.join(random.choice('01') for _ in range(tyuy * ioi))




def gestn(tyuy, ioi, dfgh=10):
    jjjk = []
    for _ in range(dfgh):
        lnm = choe(tyuy, ioi)
        if '1' in lnm:
            jjjk.append(lnm)
    return jjjk

def electents(jjjk):
    cxb = random.randint(0, len(jjjk) - 1)
    vbnm = cxb


    while vbnm == cxb:
        vbnm = random.randint(0, len(jjjk) - 1)

    return jjjk[cxb], jjjk[vbnm]


def ssovec(asdf, qwer):
    zxcv = len(asdf)
    tyhj = random.randint(1, zxcv // 2)
    bnmk = random.randint(tyhj, zxcv)

    uiop = []
    lkju = []

    for poiu in range(zxcv):
        if tyhj <= poiu < bnmk:
            uiop.append(qwer[poiu])
            lkju.append(asdf[poiu])
        else:
            uiop.append(asdf[poiu])
            lkju.append(qwer[poiu])

    return tyhj, bnmk, ''.join(uiop), ''.join(lkju)

    return tyuy, ioi


def main():



    xcvs = 'inamina.txt'

    jjjk = gestn(tyuy, ioi)
