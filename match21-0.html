import random



f = open('/content/InputFile.txt', 'r')
lines = f.readlines()
#print(lines)

FirstLine = lines[0].strip().split()
N = int(FirstLine[0])
T = int(FirstLine[1])

courses = []

for line in lines[1:]:
    courses.append(line.strip())
#print(courses)

sizeOfthePopulation = 10
RateofCrossOver = 0.7
RateofMutation = 0.01
maximumGeneration = 100

def GenerateChromosome():
    chromosome = []
    for x in range(N * T):
        Gene = random.choice('01')
        chromosome.append(Gene)
    return ''.join(chromosome)

def InitiatePopulation():
    popula = []
    for x in range(sizeOfthePopulation):
        chromosome = GenerateChromosome()
        popula.append(chromosome)
    return popula

def FitnessCalculation(chromosome):
    OverlapPenalty = 0
    ConsistencyPenalty = 0
    TimeSlots = []

    for x in range(T):
        TimeSlot = chromosome[x * N:(x + 1) * N]
        TimeSlots.append(TimeSlot)

    CourseCounts = [0] * N

    for TimeSlot in TimeSlots:
        #print(timeslot)
        CourseCount = 0
        for bitS in TimeSlot:
            CourseCount += int(bitS)

        if CourseCount > 1:
            OverlapPenalty += CourseCount - 1

        for x in range(N):
            if TimeSlot[x] == '1':
                CourseCounts[x] += 1
                #print(course_counts)
    for Count in CourseCounts:
        #print(count)
        if Count != 1:
            ConsistencyPenalty += abs(Count - 1)

    OverallPenalty = OverlapPenalty + ConsistencyPenalty
    return -OverallPenalty

def SelectionOfParents(popula, Fitness):
    TotalFitness = sum(Fitness)
    SelectionProbs = []
    for f in Fitness:
        SelectionProb = f / TotalFitness
        SelectionProbs.append(SelectionProb)
    parents = random.choices(popula, weights=SelectionProbs, k=2)
    return parents

def mutate(chromosome):
    chromosome = list(chromosome)
    #print(chromosome)

    for x in range(len(chromosome)):
        if random.random() < RateofMutation :
            if chromosome[x] == '0':
                chromosome[x] = '1'
            else:
                chromosome[x] = '0'
    return ''.join(chromosome)

def crossover(Parent1, Parent2):
    Point = random.randint(1, N * T - 1)
    Child1 = Parent1[:Point] + Parent2[Point:]
    Child2 = Parent2[:Point] + Parent1[Point:]
    return Child1, Child2





def geneticAlgo():
    popula = InitiatePopulation()
    bestChoromosome = None
    bestFit = float('-inf')

    for generation in range(maximumGeneration):
        Fitness = []
        for chromosome in popula:
            Fitnesses = FitnessCalculation(chromosome)
            Fitness.append(Fitnesses)

        if max(Fitness) > bestFit:
            bestFit = max(Fitness)
            bestChoromosome = popula[Fitness.index(max(Fitness))]

        NewPopulation = []
        while len(NewPopulation) < sizeOfthePopulation:
            Parent1, Parent2 = SelectionOfParents(popula, Fitness)

            if random.random() < RateofCrossOver:
                Child1, Child2 = crossover(Parent1, Parent2)
            else:
                Child1 = Parent1
                Child2 = Parent2

            Child1 = mutate(Child1)
            Child2 = mutate(Child2)

            NewPopulation.extend([Child1, Child2])

        popula = NewPopulation[:sizeOfthePopulation]

    return bestChoromosome, bestFit


bestChoromosome, bestFit = geneticAlgo()
print("Best Chromosome :", bestChoromosome)
print("Best Fitness :", bestFit)
