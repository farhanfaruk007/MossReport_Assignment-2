def slt_parentt(p):

def crsovr(p1, p2):
    crsovr_point = random.randint(1, len(p1) - 1)
    offsp1 = p1[:crsovr_point] + p2[crsovr_point:]
    offsp2 = p2[:crsovr_point] + p1[crsovr_point:]
    return offsp1, offsp2

def muta_te(chr, mutate_rt):

def generate__ppl(ppl_size, chr_lengthh):
