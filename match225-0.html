#Task 01:

import random

def fitness(chromosome, n, t):
    overlap_p = 0
    consistancy_p = 0


    timeslots = [chromosome[i * n:(i + 1) * n] for i in range(t)]
    for slot in timeslots:
        overlap_count = sum(slot)
        if overlap_count > 1:
            overlap_p += overlap_count - 1
    for course_idx in range(n):
        course_count = sum(timeslot[course_idx] for timeslot in timeslots)
        if course_count != 1:
            consistancy_p += abs(course_count - 1)
    total_penalty = overlap_p + consistancy_p
    return -total_penalty

def generate_chromosome(n, t):
    chromosome = [0] * (n * t)
    for course in range(n):
        slot = random.randint(0, t - 1)
        chromosome[slot * n + course] = 1
    return chromosome

def single_point_crossover(parent1, parent2):
    crossover_point = random.randint(1, len(parent1) - 1)
    child1 = parent1[:crossover_point] + parent2[crossover_point:]
    child2 = parent2[:crossover_point] + parent1[crossover_point:]
    return child1, child2

def mutate(chromosome, mutation_rate=0.1):
    for i in range(len(chromosome)):
        if random.random() < mutation_rate:
            chromosome[i] = 1 - chromosome[i]

def genetic_algorithm(n, t, courses, population_size=10, max_generations=50):
    population = [generate_chromosome(n, t) for _ in range(population_size)]
    for generation in range(max_generations):
        fitness_values = [fitness(individual, n, t) for individual in population]
        max_fitness = max(fitness_values)
        if max_fitness == 0:
            best_individual = population[fitness_values.index(max_fitness)]
            return best_individual, max_fitness
        selected_parents = random.sample(population, 2)
        parent1, parent2 = selected_parents
        child1, child2 = single_point_crossover(parent1, parent2)
        child1 = mutate(child1)
        child2 = mutate(child2)
        population[fitness_values.index(min(fitness_values))] = child1
        population[fitness_values.index(min(fitness_values))] = child2

    best_fitness = max([fitness(individual, n, t) for individual in population])
    best_individual = population[[fitness(individual, n, t) for individual in population].index(best_fitness)]
    return best_individual, best_fitness

def take_input():
    print("Enter the number of courses and timeslots (space-separated):")
    n, t = map(int, input().split())  # Ex: 3 3
    print(f"Enter the names of {n} courses (one per line):")
    courses = [input().strip() for _ in range(n)]  # Ex: CSE110, MAT110, PHY112
    return n, t, courses

if __name__ == "__main__":
    n, t, courses = take_input()
    best_schedule, best_fitness = genetic_algorithm(n, t, courses)
    print("Best Schedule:", "".join(map(str, best_schedule)))
    print("Fitness Value:", best_fitness)

import random

def two_point_crossover(parent1, parent2):
    length = len(parent1)
    point1 = random.randint(1, length - 2)
    point2 = random.randint(point1 + 1, length - 1)
    child1 = parent1[:point1] + parent2[point1:point2] + parent1[point2:]
    child2 = parent2[:point1] + parent1[point1:point2] + parent2[point2:]
    return child1, child2
