import random

def fitness(new_choromosome, N, T):

    open_loop_penalty = sum(
        max(0, sum(int(new_choromosome[t * N + i]) for i in range(N)) - 1)
        for t in range(T)
    )

    closed_system_cost = sum(
        abs(sum(int(new_choromosome[t * N + c]) for t in range(T)) - 1)
        for c in range(N)
    )

    return -(open_loop_penalty + closed_system_cost)


def generating_population(population_count, chromosome_length):
    population = []
    for _ in range(population_count):
        chromosome = ''.join(random.choices('01', k=chromosome_length))
        population.append(chromosome)
    return population
def parent_selection(population):
    parent1 = random.choice(population)
    parent2 = random.choice(population)
    while parent1 == parent2:
        parent2 = random.choice(population)
    return parent1, parent2




def Genetic_algorithm(N, T, population_size=10, max_generations=1000, mutation_rate=0.01, target_chromosome=None):

    chromosome_length = N * T
    population = generating_population(population_size, chromosome_length)

    for generation in range(max_generations):

        next_generation = []


        while len(next_generation) < population_size:

            parent1, parent2 = parent_selection(population)


            offspring1, offspring2 = crossover_1(parent1, parent2)


            offspring1 = mutation(offspring1, mutation_rate)
            offspring2 = mutation(offspring2, mutation_rate)

            next_generation.extend([offspring1, offspring2])


        population = sorted(
            next_generation,
            key=lambda x: fitness(x, N, T),
            reverse=True
        )[:population_size]


        if target_chromosome and target_chromosome in population:
            return target_chromosome, fitness(target_chromosome, N, T)


    best_chromosome = max(population, key=lambda x: fitness(x, N, T))
    return best_chromosome, fitness(best_chromosome, N, T)

def mutation(chromosome, mutation_rate=0.01):

    return ''.join(
        '1' if gene == '0' and random.random() < mutation_rate else
        '0' if gene == '1' and random.random() < mutation_rate else
        gene
        for gene in chromosome
    )


def crossover(parent1, parent2):


    min_point = len(parent1) // 4
    max_point = (3 * len(parent1)) // 4


    crossover_point = random.randint(min_point, max_point)


    offspring1 = parent1[:crossover_point] + parent2[crossover_point:]
    offspring2 = parent2[:crossover_point] + parent1[crossover_point:]

    return offspring1, offspring2

def crossover_1(parent1, parent2):

    segment_size = max(2, len(parent1) // 4)
    mutd_one = random.randint(0, len(parent1) - segment_size - 1)
    mutd_two = mutd_one + segment_size
    offspring1 = parent1[:mutd_one] + parent2[mutd_one:mutd_two] + parent1[mutd_two:]


    offspring2 = parent2[:mutd_one] + parent1[mutd_one:mutd_two] + parent2[mutd_two:]

    return offspring1, offspring2

def two_point_crossover(parent1, parent2):
    possible_indices = list(range(0, len(parent1) - 1))

    random.shuffle(possible_indices)


    pt1 = possible_indices[0]
    pt2 = random.choice([index for index in possible_indices if index > pt1])


    offset1 = parent1[:pt1] + parent2[pt1:pt2] + parent1[pt2:]
    offset2 = parent2[:pt1] + parent1[pt1:pt2] + parent2[pt2:]
