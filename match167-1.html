import random

# Fitness calculation with flexibility for constraints
def fitness_calculation(chromosome, N, T):
    overlap = 0
    consistency = 0

    # Split chromosome into time slots
    schedule = [chromosome[i * N:(i + 1) * N] for i in range(T)]



    # Calculate overlaps in time slots
    for t in schedule:
        s_courses = sum(t)
        if s_courses > 1:
            overlap += (s_courses - 1)  # Increment overlap penalty

    # Check for consistency across courses
    for i in range(N):
        number_of_courses = sum(t[i] for t in schedule)
        consistency += abs(number_of_courses - 1)  # Enforce one course per time slot

    total_penalty = overlap + consistency
    fitness = -total_penalty  # Fitness is inversely proportional to the penalty
    return fitness


# Input for the number of courses (N) and time slots (T)

    chromosome = [random.randint(0, 1) for _ in range(N * T)]

# Calculate the fitness value
fitness = fitness_calculation(chromosome, N, T)

def two_point_crossover(parent1, parent2):
    assert len(parent1) == len(parent2), "Parents must have the same length."

    length = len(parent1)
    point1 = random.randint(0, length - 2)  # Ensure space for a second point
    point2 = random.randint(point1 + 1, length - 1)  # Ensure point2 is after point1

    # Generate offspring
    offspring1 = parent1[:point1] + parent2[point1:point2] + parent1[point2:]


    offspring2 = parent2[:point1] + parent1[point1:point2] + parent2[point2:]

    return offspring1, offspring2, point1, point2
