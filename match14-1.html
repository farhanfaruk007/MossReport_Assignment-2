import random

def population(x,n,t):
  chromosomes= []
  elem= [0,1]
  for i in range(x):
     arr=[0]*(n*t)
     for j in range(len(arr)):
      y= random.choice(elem)
      arr[j] = y
     chromosomes.append(arr)
  return chromosomes

def random_selection(population):
  randomly_selected_parent= []
  for i in range(2):
    x= random.randint(0,len(population)-1)
    randomly_selected_parent.append(population[x])
  return randomly_selected_parent

def crossover(parent):
  parent1= parent[0]
  parent2= parent[1]
  offsprings = []
  x= random.randint(1,len(parent1)-1)
  offspring1= parent1[0:x] + parent2[x:len(parent1)]
  offspring2= parent2[0:x] + parent1[x:len(parent1)]
  offsprings.append(offspring1)
  offsprings.append(offspring2)
  return offsprings

def mutation(offsprings):
  for i in offsprings:
    x= random.randint(0,len(i)-1)
    if i[x] == 0:
      i[x] = 1
    else:
      i[x] = 0
  return offsprings
