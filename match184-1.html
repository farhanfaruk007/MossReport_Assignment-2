import numpy as np
import random

def fitness(population, n, t):

    f_scores = []
    for c in population:
        p = 0
        cons = 0

        for time in range(t):
            courses_scheduled = sum(c[time*n : (time+1)*n])
            if courses_scheduled > 1:
                p += courses_scheduled - 1

        for course in range(n):
            timee = sum(c[course::n])


            cons += abs(timee - 1)

        f_score = -(p + cons)
        f_scores.append(f_score)

    return np.array(f_scores)

def select(population, fit):
    total = sum(fit)
    probabilities = [f / total for f in fit]
    return population[np.random.choice(len(population), size=1, p=probabilities)[0]]

def crossover(x, y):
    n = len(x)
    point1 = random.randint(0, n-2)
    point2 = random.randint(point1+1, n-1)
    child1 = np.concatenate((x[:point1], y[point1:point2], x[point2:]))


    child2 = np.concatenate((y[:point1], x[point1:point2], y[point2:]))
    return child1, child2

def mutate(child, rate):

    for i in range(len(child)):
        if random.random() < rate:
            child[i] = 1 - child[i]
    return child

def GA(population, n, t, mutation_threshold=0.3, g=1000):

    for g in range(g):
        fit = fitness(population, n, t)
        new_population = []

        for i in range(len(population) // 2):
            parent1 = select(population, fit)
            parent2 = select(population, fit)
            child1, child2 = crossover(parent1, parent2)

            if random.random() < mutation_threshold:
                child1 = mutate(child1, 0.01)
            if random.random() < mutation_threshold:
                child2 = mutate(child2, 0.01)



            new_population.extend([child1, child2])

        population = np.array(new_population)

    fit = fitness(population, n, t)
    best_idx = np.argmax(fit)
    bestc = population[best_idx]
    best_fitness = fit[best_idx]

    print(''.join(map(str, bestc)))
    print(best_fitness)

def two_point_crossover(parent1, parent2):
    length = len(parent1)
    point1 = random.randint(1, length - 2)
    point2 = random.randint(point1 + 1, length - 1)

    child1 = parent1[:point1] + parent2[point1:point2] + parent1[point2:]
    child2 = parent2[:point1] + parent1[point1:point2] + parent2[point2:]

    return child1, child2
