import random

def population(N, T, size=10):
    pop = []
    chrom_len = N * T


    for _ in range(size):
        x = ''.join(random.choice("01") for _ in range(chrom_len))
        pop.append(x)
    return pop

def decoding(chromosome, N):
    timeslots = []
    for i in range(0, len(chromosome), N):
        timeslot = chromosome[i:i + N]
        timeslots.append(timeslot)
    return timeslots

def fitness(chrom, N, T):
    schedule = decoding(chrom, N)
    p_overlap = 0
    p_consistency = 0

    for time in schedule:
        course_overlap = sum(int(x) for x in time)
        if course_overlap > 1:
            p_overlap += (course_overlap - 1)

    course_appear = [0] * N
    for time in schedule:
        for i, course in enumerate(time):
            if course == "1":
                course_appear[i] += 1
    p_consistency = sum(abs(count - 1) for count in course_appear)

    S = p_overlap + p_consistency
    return -S

def parents(pop, fitness_values):

    adjusted_fitness = [f + abs(min(fitness_values)) for f in fitness_values]


    total_fitness = sum(adjusted_fitness)

    probabilities = [f / total_fitness for f in adjusted_fitness]

    return random.choices(pop, weights=probabilities, k=2)

def single_point_crossover(p1, p2):
    point = random.randint(1, len(p1) - 1)
    c1 = p1[:point] + p2[point:]


    c2 = p2[:point] + p1[point:]
    return c1, c2

def mutation(chrom, m_rate=0.1):
    mutate = ""
    for bit in chrom:
        if random.random() < m_rate:
            mutate += '0' if bit == '1' else '1'
        else:
            mutate += bit
    return mutate

def genetic_algo(N, T, code, generation=100, size=10):
    pop = population(N, T, size)
    best_fitness = float('-inf')
    best_chrom = None

    for _ in range(generation):
        fitness_values = [fitness(z, N, T) for z in pop]
        for i, fit in enumerate(fitness_values):
            if fit > best_fitness:
                best_fitness = fit
                best_chrom = pop[i]

        new_pop = []


        for _ in range(size // 2):
            p1, p2 = parents(pop, fitness_values)
            c1, c2 = single_point_crossover(p1, p2)
            new_pop.extend([mutation(c1), mutation(c2)])
        pop = new_pop

        if best_fitness == 0:
            break

    return best_chrom, best_fitness

N, T = 3, 3
code = ['CSE110', 'MAT110', 'PHY112']
value, f = genetic_algo(N, T, code)
print("Chromosome:", value)
print("Fitness Value:", f)

def two_point_crossover(parent1, parent2):
    length = len(parent1)
    point1 = random.randint(1, length - 2)
    point2 = random.randint(point1 + 1, length - 1)

    child1 = parent1[:point1] + parent2[point1:point2] + parent1[point2:]


    child2 = parent2[:point1] + parent1[point1:point2] + parent2[point2:]

    return child1, child2
