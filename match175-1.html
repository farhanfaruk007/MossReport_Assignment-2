def calculate_penalty(chromosome):
    overlap_penalty = 0
    consistency_penalty = 0


    for i in range(n2):
        timeslot = chromosome[i*n1:(i+1)*n1]
        #print(timeslot)
        if sum(timeslot) > 1:


            overlap_penalty += sum(timeslot) - 1


    for j in range(n1):
        course_count = sum(chromosome[j::n1])
        consistency_penalty += abs(course_count - 1)

    total_penalty = overlap_penalty + consistency_penalty
    fitness = -total_penalty
    # print(fitness)
    return fitness

def select_parents(population):
    return random.choices(population, k=2)

def crossover(parent1, parent2):
    point = random.randint(1, len(parent1) - 1)
    offspring1 = parent1[:point] + parent2[point:]


    offspring2 = parent2[:point] + parent1[point:]
    return offspring1, offspring2

def mutate(chromosome):
    mutation_rate = random.uniform(0, 1)
    for i in range(len(chromosome)):
        if random.random() < mutation_rate:
            chromosome[i] = 1 - chromosome[i]
    return chromosome
