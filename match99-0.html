#part
import random

def create_initial_population(population_size, num_courses, num_timeslots):
    population = []
    for  i in range(population_size):
        chromosome = []
        for j in range(num_courses):
            timeslot = [0] * num_timeslots
            timeslot[random.randint(0, num_timeslots - 1)] = 1
            chromosome.extend(timeslot)
        population.append(chromosome)
    return population

def calculate_fitness(chromosome, num_courses, num_timeslots):
    overlap_penalty = 0
    consistency_penalty = 0

    schedule = [chromosome[i*num_timeslots:(i+1)*num_timeslots] for i in range(num_courses)]

    for t in range(num_timeslots):
        courses_in_timeslot = sum(schedule[i][t] for i in range(num_courses))
        if courses_in_timeslot > 1:
            overlap_penalty += (courses_in_timeslot - 1)

    for i in range(num_courses):
        times_scheduled = sum(schedule[i])
        consistency_penalty += abs(times_scheduled - 1)

    fitness_score = -(overlap_penalty + consistency_penalty)
    return fitness_score

def select_parents(population, fitness_scores):
    total_fitness = sum(fitness_scores)
    selection_probabilities = [score / total_fitness for score in fitness_scores]
    parents = random.choices(population, weights=selection_probabilities, k=2)
    return parents

def perform_crossover(parent1, parent2):
    crossover_point = random.randint(1, len(parent1) - 1)
    child1 = parent1[:crossover_point] + parent2[crossover_point:]
    child2 = parent2[:crossover_point] + parent1[crossover_point:]
    return child1, child2

def apply_mutation(chromosome, mutation_rate):
    for i in range(len(chromosome)):
        if random.random() < mutation_rate:
            chromosome[i] = 1 - chromosome[i]


    return chromosome

def genetic_algorithm(num_courses, num_timeslots, courses, population_size=50, max_generations=100, mutation_rate=0.02):
    population = create_initial_population(population_size, num_courses, num_timeslots)

    for generation in range(max_generations):
        fitness_scores = [calculate_fitness(individual, num_courses, num_timeslots) for individual in population]

        if 0 in fitness_scores:
            break

        next_generation = []
        while len(next_generation) < population_size:
            parent1, parent2 = select_parents(population, fitness_scores)
            offspring1, offspring2 = perform_crossover(parent1, parent2)
            next_generation.append(apply_mutation(offspring1, mutation_rate))
            next_generation.append(apply_mutation(offspring2, mutation_rate))

        population = next_generation[:population_size]

    best_index = fitness_scores.index(max(fitness_scores))
    print(population[best_index])
    print("Fitness score:", fitness_scores[best_index])

num_courses = 3
num_timeslots = 3
courses = ["CSE110", "MAT110", "PHY112"]

genetic_algorithm(num_courses, num_timeslots, courses)
