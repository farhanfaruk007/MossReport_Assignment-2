new=open('/content/drive/MyDrive/CSE422/lab02/input.txt','r')

N=int(new2.split()[0])
T=int(new2.split()[1])
ch_len=N*T
population=[]
for i in range(10):
  population.append((random.randint(0,2,ch_len)))
def check_fitness(p):
  global T,N
  P_ovrlp=0
  P_const=0
  n=0
  k=N
  for i in range(T):

    ovrlp=0
    j=n
    while (j<k):
      ovrlp+=(p[j])
      j+=1
    P_ovrlp+=(ovrlp-1)
    n+=3
    k+=3

    const=0
    for j in range(i,T*N,T):
      const+=p[j]
    P_const+=(abs(const-1))

  fitness=(P_ovrlp+P_const)*-1
  return fitness

def offspring(p1,p2):
  global T,N
  idx=random.randint(1,(N*T)-1)
  offspring1=np.concatenate((p1[:idx],p2[idx:]))
  offspring2=np.concatenate((p2[:idx],p1[idx:]))

  return offspring1,offspring2

def selection(population):
  global T,N
  idx=random.randint(0,len(population)-1)
  return population[idx]

def mutation(p):
  global T,N
  idx=random.randint(0,(N*T)-1)


  if p[idx]==0:
    p[idx]=1
  else:
    p[idx]=0
  return p

def GA(population):


  global T,N
  fitness_dict={}
  for i in range(10):
    j=0
    while j<len(population):
      fitness_dict[check_fitness(population[j])]=population[j]
      j+=1

    if max(fitness_dict.keys())==0:
      return fitness_dict[max(fitness_dict.keys())],max(fitness_dict.keys())
    new_gen=[]
    for i in range(10):
      p1=selection(population)
      p2=selection(population)
      offspring1,offspring2=offspring(p1,p2)
      offspring1=mutation(offspring1)
      offspring2=mutation(offspring2)
      new_gen.append(offspring1)
      new_gen.append(offspring2)
    population=new_gen
  max_fit=max(fitness_dict.keys())
  fit_ch=fitness_dict[max_fit]
  return fit_ch,max_fit


fit_ch,fit_score=GA(population)
