def gp(d,a,b):
  p_o_=[]
  for i in range(d):
    chr=[random.choice([0,1]) for i in range(a*b)]
    p_o_.append(chr)
  return p_o_

def c_ftnss(chr,y,t):
  ov=0
  cn=0
  for b in range(t):
    cou_t=chr[b*y:(b+1)*y]
    ov+=abs(sum(cou_t)-1) if sum(cou_t)!=1 else 0
  for k in range(y):
    t_lst=[]
    t_sch=sum(t_lst)
    t_lst=[chr[x*y+k] for x in range (t)]
    if sum(t_lst)!=1:
      cn+=abs(sum(t_lst)-1)
  t_plty=ov+cn
  return -t_plty



def guard(p_o_,ft_y):
  m_ft=min(ft_y)
  ft_ad=[f-m_ft+1 for f in ft_y]
  ttl=sum(ft_ad)
  pty_r=[ff/ttl for ff in ft_ad]
  grip=random.choices(range(len(p_o_)),pty_r,k=2)
  return p_o_[grip[0]],p_o_[grip[1]]

def crss_ov(pn1,pn2):
  s=random.randint(1,len(pn1)-2)
  of_one=pn1[:s]+pn2[s:]
  of_two=pn2[:s]+pn1[s:]
  return of_one,of_two

def mtat(chr,mtat=0.1):
  for j in range(len(chr)):
    if random.random()<mtat:
      chr[j]=1-chr[j]
