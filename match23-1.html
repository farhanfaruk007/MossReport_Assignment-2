import random

class GenAlgo:
    def __init__(self, num, tym, crs, p_s=50, m_r=0.01, max=50):
        self.num = num
        self.tym = tym
        self.crs = crs
        self.p_s= p_s
        self.m_r = m_r
        self.max = max
        self.ppl = self.genPop()

    def genPop(self):
        return [[random.randint(0, 1) for _ in range(self.num * self.tym)] for _ in range(self.p_s)]

    def fitFunc(self, crsm):
        pOver = 0
        pCons = 0
        res = [crsm[i * self.num:(i + 1) * self.num] for i in range(self.tym)]
        for tymslot in res:
            pOver += max(0, sum(tymslot) - 1)
        crsC = [sum(res[j][i] for j in range(self.tym)) for i in range(self.num)]
        pCons += sum(abs(count - 1) for count in crsC)
        return -(pOver + pCons)

    def parent_sel(self, fitt):
        totalfit = sum(fitt)
        pro = [f / totalfit for f in fitt]
        return random.choices(self.ppl, pro, k=2)

    def cross(self, p1, p2):
        mark = random.randint(1, len(p1) - 1)
        baby1 = p1[:mark] + p2[mark:]
        baby2 = p2[:mark] + p1[mark:]
        return baby1, baby2

    def mutation(self, crsm):
        return [
            1 - gn if random.random() < self.m_r else gn
            for gn in crsm
        ]

    def optimizationFunc(self):
        soln = None
        goodFit = float('-inf')

        for gen in range(self.max):
            fitt = [self.fitFunc(crsm) for crsm in self.ppl]

            index = fitt.index(max(fitt))
            if fitt[index] > goodFit:
                goodFit = fitt[index]
                soln = self.ppl[index]

            newppl = []
            for _ in range(self.p_s // 2):
                p1, p2 = self.parent_sel(fitt)
                baby1, baby2 = self.cross(p1, p2)
                newppl.append(self.mutation(baby1))
                newppl.append(self.mutation(baby2))

            self.ppl = newppl

        return soln, goodFit

def main(input_file):
    with open(input_file, 'r') as file:
        lines = file.readlines()

    num, tym = map(int, lines[0].strip().split())
    crs = [line.strip() for line in lines[1:]]

    if tym < num:
        print("Error: Number of timeslots (T) must be greater than or equal to the number of courses (N).")
        return

    check = GenAlgo(num, tym, crs)
    result, ftns = check.optimizationFunc()

    print("Optimal Schedule (Binary String):", ''.join(map(str, result)))
    print("Fitness Value:", ftns)


if __name__ == "__main__":
    input_file = "input.txt"
    main(input_file)
