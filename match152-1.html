  choose2 = random.randint(0,len(initialpopulation)-1)

  parent1 = initialpopulation[choose1]
  parent2 = initialpopulation[choose2]

def crossover(parent1, parent2):

    crosspoint = random.randint(1, len(parent1) - 1)
    child1 = parent1[:crosspoint] + parent2[crosspoint:]
    child2 = parent2[:crosspoint] + parent1[crosspoint:]

    return child1,child2

def mutation(chromsome,mutprob=0.01):

  for i in range(len(chromsome)):
