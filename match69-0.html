import math
import random

def overlap_penalty(t_slots,x,y):
    arr=[]
    temp=0

    for i in range(y):
        in_slot=t_slots[i*x:(i+1)*y]
        count=max(in_slot.count("1")-1,0)
        temp+=count
    return temp

def consistency_panelty(t_slots,x,y):
    count=0

    for i in range(x):
        count_p = 0
        for j in range(y):
            count_p+=int(t_slots[j*x+i])
        count+=abs(count_p-1)
    return count

def calculate_fitness(t_slots,x,y):
    p= -(overlap_penalty(t_slots,x,y)+consistency_panelty(t_slots,x,y))
    return p

def create_population(X,Y,population_size):
    people_list=[]

    for j in range(population_size):
        people = ""
        for i in range(X*Y):
            people+=random.choice('01')
        people_list.append(people)
    return people_list

def parents_selection(population):
    return random.choice(population),random.choice(population)

def crossover(parent1,parent2,x,y):
    point=random.randint(0,x*y-2)
    point2=random.randint(point+1,x*y-1)
    child1=parent1[0:point]+parent2[point:point2]+parent1[point2:]


    child2=parent2[0:point]+parent1[point:point2]+parent2[point2:]
    return child1,child2

def mutation(child,x,y):
    point=random.randint(1,x*y-1)

    list1=list(child)
    if list1[point]=="1":
        list1[point]="0"
    else:
        list1[point]="1"
    return "".join(list1)


def genetic_algorithm(x,y,population_size,iterations):
    population=create_population(x,y,population_size)
    lowest_fitness=-math.inf
    best_slot=None

    for _ in range(iterations):
        new_population=[]

        for _ in range(population_size//2):
            parent1,parent2=parents_selection(population)
            child1,child2=crossover(parent1,parent2,x,y)
            new_population.extend([mutation(child1,x,y),mutation(child2,x,y)])
        population=new_population

        for slots in population:
            fitess_value=calculate_fitness(slots,x,y)

            if fitess_value>lowest_fitness:
                lowest_fitness = fitess_value
                best_slot = slots

            if lowest_fitness==0:
                break
    return lowest_fitness,best_slot
