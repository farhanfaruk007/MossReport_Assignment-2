import random
num_of_lines = input()
courses_list = []

for i in range(int(number_list[0])):
  courses_name=input()

#############First Part

def fitness_function(chromosome, course, Timeslot):


    consistancy_penalty = 0
    overlap_penalty = 0
    course_list = [0]*course
    for k in range(0,(course*Timeslot)):
        if chromosome[k] == str(1):


            course_list[k%course]+=1
    for j in range(0, len(course_list)):
        if course_list[j]!=1:
            consistancy_penalty+= abs(course_list[j]) #because initialize in 0
    end_point=len(chromosome)//Timeslot
    #print(end_point)
    for i in range(0, end_point):
        slot = chromosome[(i*course): ((i+1)*course)]
        sum = 0
        for j in range(len(slot)):
            sum+=int(slot[j])
        if sum>1:
            overlap_penalty+= (sum-1)


    penalty = (consistancy_penalty+overlap_penalty)*(-1)
    return penalty

def initializationOfPopulation(number_of_population,course, Timeslot):
    population= []
    count=0
    while count<number_of_population:
        count+=1
        chromosome = [0]*(course*Timeslot)
        chromosome_string = ""


        for j in range(0, Timeslot):
            slot_generation = random.randint(0, course-1)
            chromosome[j*course+slot_generation] = 1
        for k in range(0, course):
            slot_generation = random.randint(0, Timeslot-1)
            chromosome[slot_generation*course+k] = 1
        for m in chromosome:
            chromosome_string += str(m)
        population.append(chromosome_string)

    return population




def chromosomeCrossover(par1, par2):
    crossover_point = random.randint(1, len(par1) - 1)

    part1_from_par1 = par1[:crossover_point]
    part2_from_par2 = par2[crossover_point:]

    part1_from_par2 = par2[:crossover_point]
    part2_from_par1 = par1[crossover_point:]

    child1 = part1_from_par1 + part2_from_par2
    child2 = part1_from_par2 + part2_from_par1



    return child1, child2


def mutationOfChromosome(chromosome, probability):
    chromosome_list = list(chromosome)
    for i in range(0, len(chromosome_list)):
        random_num= random.random()
        if random_num < probability:
            chromosome_list[i] = str(0)

    for j in range(0, Timeslot):
        slot_seg = chromosome_list[j*course : (j+1)*course]
        temp=0
        for searching_one in slot_seg:
          if searching_one=="1":
            temp+=1
        if temp== 0:
            randomized_course = random.randint(0, course-1)
            chromosome_list[j*course+randomized_course] = str(1)

    mutated_chromosome = ""
    for j in range(len(chromosome_list)):
        mutated_chromosome+=chromosome_list[j]
    return mutated_chromosome

def geneticAlgorithm(course, Timeslot, size_of_population, probability):
    popul = initializationOfPopulation(size_of_population, course, Timeslot)
    highest_fitness_value = float('-inf')
    fittest_chromosome = ""




    for i in range(0, 25):
        newPopulation = []
        for j in range(0, 10):
            par1, par2 = random.sample(popul, 2)
            ch1, ch2 = chromosomeCrossover(par1, par2)
            ch1 = mutationOfChromosome(ch1, probability)
            ch2 = mutationOfChromosome(ch2, probability)
            newPopulation.append(ch1)
            newPopulation.append(ch2)
        popu_dict = {}
        for child in newPopulation:
            popu_dict[child] = fitness_function(child, course, Timeslot)
        newPopulation = []
        sorted_key_li = sorted(popu_dict, key = popu_dict.get, reverse = True)
        for child in range(0, len(sorted_key_li)):
            newPopulation.append(sorted_key_li[child])
        popul = newPopulation[:size_of_population]
        for k in popul:
            fitness_value = fitness_function(k, course, Timeslot)
            if fitness_value > highest_fitness_value:
                highest_fitness_value = fitness_value
                fittest_chromosome = k
    return fittest_chromosome, highest_fitness_value

fittest_chromosome, highest_fitness_value = geneticAlgorithm(course,Timeslot, 10, 0.01)

print("Part 1:")
print("Fittest chromosome:",fittest_chromosome)
print("Fitness value:",highest_fitness_value)
