import random
def gen_p(n,t,ps):
    p=[]
    chrom_len=n*t
    for i in range(ps):
        chrom=[]
        for j in range(chrom_len):
            chrom.append(random.randint(0,1))
        p.append(chrom)
    return p
def calc_fit(chrom,n,t):
    slots=[]
    for i in range(0,len(chrom),n):
        slots.append(chrom[i:i+n])
    overlap_pen=0
    consistency_pen=0
    for slot in slots:
        overlap = sum(slot)-1
        if overlap>0:
            overlap_pen+=overlap
    course_counts=[0]*n
    for slot in slots:
        for i in range(n):
            course_counts[i]+=slot[i]
    for count in course_counts:
        consistency_pen += abs(count-1)
    return -(overlap_pen+consistency_pen)
def crossover(p1,p2):
    cp=random.randint(1,len(p1)-1)
    c1=p1[:cp]+p2[cp:]
    c2=p2[:cp]+p1[cp:]
    return c1,c2
def mutate(chrom,mr):


    for i in range(len(chrom)):
        if random.random()<mr:
            chrom[i]=1-chrom[i]
    return chrom
def select_parents(p):
    idx1=random.randint(0,len(p)-1)
    idx2=random.randint(0,len(p)-1)
    return p[idx1],p[idx2]
def gen_algo(n,t,ps,max_gen):
    p=gen_p(n,t,ps)
    for gen in range(max_gen):
        fit_scores=[]
        for chrom in p:
            fit_scores.append(calc_fit(chrom,n,t))


        max_fit=max(fit_scores)
        if max_fit==0:
            sol=p[fit_scores.index(max_fit)]
            print(f"Solution in Gen {gen}")
            return sol,max_fit
        next_gen=[]
        for i in range(ps//2):
            p1,p2=select_parents(p)
            c1,c2=crossover(p1,p2)
            next_gen.append(mutate(c1,mrate))
            next_gen.append(mutate(c2,mrate))
        p=next_gen
    best_idx=fit_scores.index(max(fit_scores))
    return p[best_idx], max(fit_scores)
n,t=map(int, input().split())
courses=[]
for i in range(n):
    course=input(f"Course{i+1}:")
    courses.append(course)
ps=100
mrate=0.1


max_gen=min(10**n,20000)
best_sched,fit_val=gen_algo(n,t,ps,max_gen)
print("Gen Solved:",''.join(map(str,best_sched)))
print("Fit Val:",fit_val)
