#TASK-01
import random

def chromosome(N, T):
    arr_len = N * T
    cases = "01"
    chromosome = ""
    for i in range(arr_len):
        chromosome += random.choice(cases)
    return chromosome

# Fitness function
def fitness_function(chromosome, N, T):
    overlap_penalty = 0
    consistency_penalty = 0
    timeslots =[]
    for i in range(T):
        start = i * N
        end = (i+1)* N
        slot = chromosome[start:end]
        timeslots.append(slot)

    #overlap penalty
    for slot in timeslots:
        count = slot.count('1')
        if count > 1:
            penalty = count - 1
            overlap_penalty += penalty

    #Consistency penalty
    for i in range(N):
        count1 = 0
        for slot in timeslots:
            if slot[i] == '1':
                count1 += 1
        penalty1 = abs(count1 - 1)
        consistency_penalty += penalty1

    total_penalty = overlap_penalty + consistency_penalty
    fitness = -total_penalty
    return fitness

def select_parents(population):
    return random.sample(population, 2)

# Single-point crossover
def crossover(parent1, parent2):
    crossover_point = random.randint(1, len(parent1) - 1)
    offspring1 = parent1[:crossover_point] + parent2[crossover_point:]


    offspring2 = parent2[:crossover_point] + parent1[crossover_point:]
    return offspring1, offspring2

def mutate(chromosome, mutation_rate=0.01):
    mutated_chromosome = []
    for char in chromosome:
        if random.random() > mutation_rate:
            mutated_chromosome.append(char)
        else:
            if char=="0":
                mutated_chromosome += "1"
            else:
                mutated_chromosome += "0"

    return ''.join(mutated_chromosome)

def genetic_algorithm(N, T, iterations=200):
    best_fitness = float('-inf')
    best_chromosome = None
    population_size = 50
    population = []

    # Generate initial population
    for i in range(population_size):
        chromo = chromosome(N, T)
        population.append(chromo)

    for i in range(iterations):
        population = sorted(population, key=lambda x: fitness_function(x, N, T), reverse=True)
        next_population = population[:population_size//2]

        while len(next_population) < population_size:
            parent1, parent2 = select_parents(population)
            offspring1, offspring2 = crossover(parent1, parent2)
            next_population.extend([mutate(offspring1), mutate(offspring2)])

        population = next_population
        current_best = max(population, key=lambda x: fitness_function(x, N, T))
        current_fitness = fitness_function(current_best, N, T)

        if current_fitness > best_fitness:
            best_fitness = current_fitness
            best_chromosome = current_best

    return best_chromosome, best_fitness

#Input and output
courses = ["CSE110", "MAT110", "PHY112"]
N = len(courses)
T = 3

best_schedule, fitness_value = genetic_algorithm(N, T)
print("Best Schedule:", best_schedule)
print("Fitness Value:", fitness_value)
