import random

class GENE_ALGO:
    def __init__(self, N = 3, T = 3, POP_SIZE = 100, GENES_AT_MAX = 1000, MUT_RATE = 0.05):
        self.N = N

        self.T = T

        self.POP_SIZE = POP_SIZE

        self.GENES_AT_MAX = GENES_AT_MAX

        self.MUT_RATE = MUT_RATE

    def Pop_create(self):
        pop = []

        for k in range(self.POP_SIZE):
            chrom = ''.join(random.choice('01') for k in range(self.N * self.T))
            pop.append(chrom)
        return pop

    def fitness(self, chrom):
        ovr_pnlty = 0
        con_pnlty = 0


        segment_of_T = [chrom[k * self.N:(k + 1) * self.N] for k in range(self.T)]


        for f in segment_of_T:


            courses_number = sum(int(biits) for biits in f)
            if courses_number > 1:
                ovr_pnlty += courses_number - 1


        sche_C = [0] * self.N

        for f in segment_of_T:
            for k, biits in enumerate(f):
                sche_C[k] += int(biits)

        for m in sche_C:
            if m != 1:
                con_pnlty + = abs(m - 1)

        val_ft = -(ovr_pnlty + con_pnlty)
        return val_ft

    def Par_ents(self, pop, fitness_score):
        fitness_cal = sum(fitness_score)
        prob_select = [a_score / fitness_cal for a_score in fitness_score]


        c_parent = random.choices(pop, prob_select, k=2)
        return c_parent

    def crossing_over(self, P1, P2):
        crossing_pnts = random.randint(1, len(P1) - 1)
        c_1 = P1[:crossing_pnts] + P2[crossing_pnts:]
        c_2 = P2[:crossing_pnts] + P1[crossing_pnts:]
        return c_1, c_2



    def Mut_a(self, chrom):
        chrom_lst = list(chrom)
        for k in range(len(chrom_lst)):
            if random.random() < self.MUT_RATE:
                chrom_lst[k] = '1' if chrom_lst[k] == '0' else '0'
        return ''.join(chrom_lst)

    def gen_algo(self):
        pop = self.Pop_create()
        chrom_opt = None
        fit_opt = float('-inf')

        for gene in range(self.GENES_AT_MAX):
            fitness_score = [self.fitness(chrom) for chrom in pop]

            if max(fitness_score) > fit_opt:
                fit_opt = max(fitness_score)
                chrom_opt = pop[fitness_score.index(fit_opt)]

            NEW_pop = []
            while len(NEW_pop) < self.POP_SIZE:
                P1, P2 = self.Par_ents(pop, fitness_score)
                c_1, c_2 = self.crossing_over(P1, P2)
                c_1 = self.Mut_a(c_1)
                c_2 = self.Mut_a(c_2)
                NEW_pop.extend([c_1, c_2])

            pop = NEW_pop[:self.POP_SIZE]

        return chrom_opt, fit_opt

N = 3
T = 3
gen_algo = GENE_ALGO (N, T)


gen_algo.POP_SIZE = 100
gen_algo.GENES_AT_MAX = 1500
gen_algo.MUT_RATE = 0.05


BesT_ChromoSome, BesT_FitNess = gen_algo.gen_algo()

print('BEST Chromosome:',BesT_ChromoSome)
