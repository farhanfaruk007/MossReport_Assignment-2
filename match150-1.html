def evaluate_fitness(chromosome):
    overlap_count = 0
    consistency_count = 0
    course_schedule_count = [0] * num_courses
    for i in range(0, len(chromosome), num_courses):
        slot_sum = sum(chromosome[i: i+num_courses])


        if slot_sum > 1:
            overlap_count += (slot_sum - 1)
    for i in range(len(chromosome)):
        idx = i % num_courses
        course_schedule_count[idx] += chromosome[i]
    for count in course_schedule_count:


        consistency_count += abs(count - 1)

    return -1 * (overlap_count + consistency_count)

def crossover_single_point(parent1, parent2):
    idx = random.randint(1, len(parent1)-2)
    offspring1 = parent1[:idx] + parent2[idx:]


    offspring2 = parent2[:idx] + parent1[idx:]
    return [offspring1, offspring2]

def apply_mutation(offspring1, offspring2):
    idx1 = random.randint(0, len(offspring1)-1)
    idx2 = random.randint(0, len(offspring2)-1)
    offspring1[idx1] = 1 - offspring1[idx1]
    offspring2[idx2] = 1 - offspring2[idx2]
    return [offspring1, offspring2]
