#Task1
import random


def initialize_population(population_size, num_courses, num_timeslots):
    population = []
    for _ in range(population_size):
        chromosome = ''.join(random.choice('01') for _ in range(num_courses * num_timeslots))
        while '1' not in chromosome:
            chromosome = ''.join(random.choice('01') for _ in range(num_courses * num_timeslots))
        population.append(chromosome)
    return population


def calculate_fitness(chromosome, num_courses, num_timeslots):
    overlap_penalty = 0
    consistency_penalty = 0
    timeslot_course_mapping = [chromosome[i * num_courses:(i + 1) * num_courses] for i in range(num_timeslots)]


    for timeslot in timeslot_course_mapping:
        scheduled_courses = sum(int(bit) for bit in timeslot)
        if scheduled_courses > 1:
            overlap_penalty += (scheduled_courses - 1)


    for course_index in range(num_courses):
        course_schedule_count = sum(int(timeslot_course_mapping[t][course_index]) for t in range(num_timeslots))
        if course_schedule_count != 1:
            consistency_penalty += abs(course_schedule_count - 1)

    total_penalty = overlap_penalty + consistency_penalty
    return -total_penalty


def select_parents(population, fitness_scores):
    total_fitness = sum(fitness_scores)
    selection_probabilities = [score / total_fitness for score in fitness_scores]
    selected_parents = random.choices(population, weights=selection_probabilities, k=2)
    return selected_parents


def perform_crossover(parent1, parent2):
    crossover_point = random.randint(1, len(parent1) - 1)
    child1 = parent1[:crossover_point] + parent2[crossover_point:]
    child2 = parent2[:crossover_point] + parent1[crossover_point:]
    return child1, child2


def apply_mutation(chromosome, mutation_probability):
    mutated_chromosome = ''.join(
        bit if random.random() > mutation_probability else str(1 - int(bit))
        for bit in chromosome
    )
    return mutated_chromosome


def run_genetic_algorithm(num_courses, num_timeslots, course_list, population_size=50, max_generations=1000, mutation_probability=0.01):
    population = initialize_population(population_size, num_courses, num_timeslots)
    best_chromosome = None
    best_fitness_score = float('-inf')

    for generation in range(max_generations):
        fitness_scores = [calculate_fitness(chromosome, num_courses, num_timeslots) for chromosome in population]
        for i, score in enumerate(fitness_scores):
            if score > best_fitness_score:
                best_fitness_score = score
                best_chromosome = population[i]

        if best_fitness_score == 0:
            break

        next_generation = []
        while len(next_generation) < population_size:
            parent1, parent2 = select_parents(population, fitness_scores)
            child1, child2 = perform_crossover(parent1, parent2)
            child1 = apply_mutation(child1, mutation_probability)
            child2 = apply_mutation(child2, mutation_probability)
            next_generation.extend([child1, child2])

        population = next_generation[:population_size]

    return best_chromosome, best_fitness_score


# Example usage
num_courses = 3
num_timeslots = 3
course_list = ["CSE110", "MAT110", "PHY112"]

best_schedule, best_fitness = run_genetic_algorithm(num_courses, num_timeslots, course_list)
print("Best Schedule:", best_schedule)
print("Fitness Value:", best_fitness)

import random

def initialize_population(population_size, num_classes, num_slots):
    population = []
    for _ in range(population_size):
        genome = ''.join(random.choice('01') for _ in range(num_classes * num_slots))
        while '1' not in genome:
            genome = ''.join(random.choice('01') for _ in range(num_classes * num_slots))
        population.append(genome)
    return population

def select_parents_randomly(population):
    return random.sample(population, 2)

def perform_two_point_crossover(parent1, parent2):
    g_length = len(parent1)

    crossover_point1 = random.randint(1, g_length - 2)
    crossover_point2 = random.randint(crossover_point1 + 1, g_length - 1)

    child1 = parent1[:crossover_point1] + parent2[crossover_point1:crossover_point2] + parent1[crossover_point2:]


    child2 = parent2[:crossover_point1] + parent1[crossover_point1:crossover_point2] + parent2[crossover_point2:]

    return child1, child2

# Example usage
num_classes = 3
num_slots = 3
population_size = 10

population = initialize_population(population_size, num_classes, num_slots)

parent1, parent2 = select_parents_randomly(population)

child1, child2 = perform_two_point_crossover(parent1, parent2)

print("Parent 1:", parent1)
print("Parent 2:", parent2)
print("Offspring 1:", child1)
print("Offspring 2:", child2)
