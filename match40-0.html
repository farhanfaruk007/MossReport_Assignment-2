#TASK-01
import random

def chromosome(N, T):
    len = N * T
    cases = "01"
    chromosome = ""
    for i in range(len):
        chromosome += random.choice(cases)
    return chromosome


def fitnessFunc(chromosome, N, T):
    openalty = 0
    conPenalty = 0
    timeslots =[]
    for i in range(T):
        start = i * N
        end = (i+1)* N
        slot = chromosome[start:end]
        timeslots.append(slot)


    for slot in timeslots:
        count = slot.count('1')
        if count > 1:
            openalty = count - 1
            openalty += openalty

    for i in range(N):
        count1 = 0
        for slot in timeslots:
            if slot[i] == '1':
                count1 += 1
        penalty1 = abs(count1 - 1)
        conPenalty += penalty1

    total_penalty = openalty + conPenalty
    fitness = -total_penalty
    return fitness

def select_parents(population):
    return random.sample(population, 2)

# Single-point
def crossover(pc1, pc2):
    coPoint = random.randint(1, len(pc1) - 1)
    ch1 = pc1[:coPoint] + pc2[coPoint:]


    ch2 = pc2[:coPoint] + pc1[coPoint:]
    return ch1, ch2

def mutate(chromosome, mutation_rate=0.02):
    mutated_ch = []
    for char in chromosome:
        if random.random() > mutation_rate:
            mutated_ch.append(char)
        else:
            if char=="0":
                mutated_ch += "1"
            else:
                mutated_ch += "0"

    return ''.join(mutated_ch)

def gA(N, T, ite=150):
    best_fitness = float('-inf')
    best_chromosome = None
    popSize = 25
    population = []
    for i in range(popSize):
        chromo = chromosome(N, T)
        population.append(chromo)

    for i in range(ite):
        population = sorted(population, key=lambda x: fitnessFunc(x, N, T), reverse=True)
        next_population = population[:popSize//2]

        while len(next_population) < popSize:
            pc1, pc2 = select_parents(population)
            ch1, ch2 = crossover(pc1, pc2)
            next_population.extend([mutate(ch1), mutate(ch2)])

        population = next_population
        current_best = max(population, key=lambda x: fitnessFunc(x, N, T))
        current_fitness = fitnessFunc(current_best, N, T)

        if current_fitness > best_fitness:
            best_fitness = current_fitness
            best_chromosome = current_best

    return best_chromosome, best_fitness

best_schedule, fitness_value = gA(N, T)
print("Best Schedule:", best_schedule)
